3. 安装JDK

  • 对于免安装压缩格式的版本，解压
$ tar -zxvf jdk-8u162-linux-x64.tar.gz -C /opt/java/
  • 对于rpm二进制格式的版本，执行
$ rpm -ivh jdk-8u131-linux-x64.rpm

4. 配置环境变量

方法一（最佳实践）

  • 创建新文件并打开
[root@devops-vm-0 opt]# vi /etc/profile.d/java.sh
  • 输入以下内容并保存
# 输入内容

# java

export JAVA_HOME=/opt/java/jdk1.8.0_162

export JRE_HOME=/opt/java/jdk1.8.0_162/jre

export PATH=$PATH:$JAVA_HOME/bin:$JRE_HOME/bin

export CLASSPATH=.:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar:$JRE_HOME/lib

# 保存退出

:wq!
  • 立即生效环境变量
[root@devops-vm-0 opt]# source /etc/profile.d/java.sh
  • 自动化脚本
$ rm -rf /etc/profile.d/java.sh

$ touch /etc/profile.d/java.sh

$ echo '# java' > /etc/profile.d/java.sh

$ echo 'export JAVA_HOME=/opt/java/jdk1.8.0_162' >> /etc/profile.d/java.sh

$ echo 'export JRE_HOME=/opt/java/jdk1.8.0_162/jre' >> /etc/profile.d/java.sh

$ echo 'export PATH=$PATH:$JAVA_HOME/bin:$JRE_HOME/bin' >> /etc/profile.d/java.sh

$ echo 'export CLASSPATH=.:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar:$JRE_HOME/lib' >> /etc/profile.d/java.sh

$ echo '' >> /etc/profile.d/java.sh

6. 自动化脚本

  • install-and-setup-java.sh
#!/bin/bash



# /etc/profile.d目录

ETC_PROFILE_D='/etc/profile.d'

echo "ETC_PROFILE_D=$ETC_PROFILE_D"



# java.sh文件路径

JAVA_ENV_FILE="$ETC_PROFILE_D/java.sh"

echo "JAVA_ENV_FILE=$JAVA_ENV_FILE"



# 创建java环境变量文件

function createJavaEnv() {

        echo "创建java环境变量文件：$1"

        rm -rf $1

        touch $1

        echo '#!/bin/bash' > $1

        echo '' >> $1

        echo '# java' >> $1

        echo 'export JAVA_HOME=/opt/java/jdk1.8.0_162' >> $1

        echo 'export JRE_HOME=/opt/java/jdk1.8.0_162/jre' >> $1

        echo 'export PATH=$PATH:$JAVA_HOME/bin:$JRE_HOME/bin' >> $1

        echo 'export CLASSPATH=.:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar:$JRE_HOME/lib' >> $1

        echo '' >> $1

}



# 执行脚本

createJavaEnv $JAVA_ENV_FILE

source $JAVA_ENV_FILE
