CentOS7安装与配置Tomcat

前置条件

JDK已安装完成，并配置环境变量。
JAVA_HOME=/opt/java/jdk1.8.0_162
JRE_HOME=/opt/java/jdk1.8.0_162/jre
CLASSPATH=.:%JAVA_HOME%/lib/dt.jar:%JAVA_HOME%/lib/tools.jar:JRE_HOME/lib
PATH=$PATH:%JAVA_HOME%/bin

下载tomcat

[root@devops-vm-0 ~]# mkdir -p /opt/tomcat

[root@devops-vm-0 ~]# cd /opt/tomcat

[root@devops-vm-0 tomcat]# wget http://mirrors.shu.edu.cn/apache/tomcat/tomcat-9/v9.0.6/bin/apache-tomcat-9.0.6.tar.gz

...

[root@devops-vm-0 tomcat]# tar -zxvf apache-tomcat-9.0.6.tar.gz

启动tomcat

进入tomcat的bin目录，执行./startup.sh命令即可。

[root@devops-vm-0 tomcat]# cd /opt/tomcat/apache-tomcat-9.0.6/bin

[root@devops-vm-0 bin]# ./startup.sh 

Using CATALINA_BASE:   /opt/tomcat/apache-tomcat-9.0.6

Using CATALINA_HOME:   /opt/tomcat/apache-tomcat-9.0.6

Using CATALINA_TMPDIR: /opt/tomcat/apache-tomcat-9.0.6/temp

Using JRE_HOME:        /opt/java/jdk1.8.0_162/jre

Using CLASSPATH:       /opt/tomcat/apache-tomcat-9.0.6/bin/bootstrap.jar:/opt/tomcat/apache-tomcat-9.0.6/bin/tomcat-juli.jar

Tomcat started.

[root@devops-vm-0 bin]#
